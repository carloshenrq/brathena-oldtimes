		next;
		mes "[Aibakthing]";
		mes "Hoo! Hahh! Hoo! Hahh!";
		mes "Nenhum homem vive pode resistir";
		mes "ao convite desta bela terra.";
		mes "Habitantes de toda Rune-Midgard";
		mes "são sempre bem-vindos!";
		close;
	case 2:
		mes "[Aibakthing]";
		mes "Ah, você tem que pegar 10.000 Zeny";
		mes "se quiser visitar Ayothaya.";
		mes "Quando estiver pronto,";
		mes "poderemos ir!";
		next;
		if (select("Estou pronto, vamos lá!:Não.") == 1) {
			if (Zeny > 9999) {
				mes "[Aibakthing]";
				mes "Hoo! Hah! Hoo! Hah!";
				mes "Vamos partir! Voltar para";
				mes "a minha bela Ayothaya!";
				close2;
				set zeny,zeny-10000;
				warp "ayothaya",149,71;
				end;

			mes "[Aibakthing]";
			mes "Lamento, mas você não possui 10.000 zeny para viajar para Ayothaya. Tal preço não é nada comparado com as experiências que esperam por você!";
			close;

		mes "[Aibakthing]";
		mes "Alguma vez você já sonhou";
		mes "com um belo lugar cheio";
		mes "de serenidade espiritual? Você";
		mes "deve ter sonhado com";
		mes "Ayothaya, meu amigo.";
		close;
	case 3:
		mes "[Aibakthing]";
		mes "Alguma vez você já sonhou";
		mes "com um belo lugar cheio";
		mes "de serenidade espiritual? Você";
		mes "deve ter sonhado com";
		mes "Ayothaya, meu amigo.";
		close;
