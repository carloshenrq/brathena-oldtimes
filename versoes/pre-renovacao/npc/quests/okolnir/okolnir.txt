	if(strnpcinfo(2) == "okolnir_time1 SE") end;
	if(compare(strnpcinfo(2),"aru")) set .cas$,"aru";
	else set .cas$,"sch";
	if(compare(strnpcinfo(2),"01")) set .casn$,"01";
	else if(compare(strnpcinfo(2),"02")) set .casn$,"02";
	else if(compare(strnpcinfo(2),"03")) set .casn$,"03";
	else if(compare(strnpcinfo(2),"04")) set .casn$,"04";
	else if(compare(strnpcinfo(2),"05")) set .casn$,"05";
	if (getd ("$siz_"+.cas$+.casn$+"_on") == 2) {
		initnpctimer;
	}
	end;

OnEnable:
	if(strnpcinfo(2) == "okolnir_time1 SE") end;
	if(compare(strnpcinfo(2),"aru")) set .cas$,"aru";
	else set .cas$,"sch";
	if(compare(strnpcinfo(2),"01")) set .casn$,"01";
	else if(compare(strnpcinfo(2),"02")) set .casn$,"02";
	else if(compare(strnpcinfo(2),"03")) set .casn$,"03";
	else if(compare(strnpcinfo(2),"04")) set .casn$,"04";
	else if(compare(strnpcinfo(2),"05")) set .casn$,"05";
	setd "$gqse_"+.cas$+.casn$+"_time",0;
	initnpctimer;
	end;

Onreset:
	setd "$siz_"+.cas$+.casn$+"_on",0;
	setd "$gqse_"+.cas$+.casn$+"_time",0;
	stopnpctimer;
	end;

OnTimer3600000:
	if (getd ("$gqse_"+.cas$+.casn$+"_time") < 12) {
		setd "$gqse_"+.cas$+.casn$+"_time",getd ("$gqse_"+.cas$+.casn$+"_time") +1;
		initnpctimer;
	}
	else if (getd ("$gqse_"+.cas$+.casn$+"_time") == 12) {
		setd "$siz_"+.cas$+.casn$+"_on",0;
		setd "$gqse_"+.cas$+.casn$+"_time",0;
		enablenpc "Dama dos Desejos#"+.cas$+"g_cas"+.casn$;
		enablenpc "Piamette#"+.cas$+"g_cas"+.casn$;
		hideoffnpc "Dama dos Desejos#"+.cas$+.casn$+"_boss";
		stopnpctimer;
	}
	end;
}
