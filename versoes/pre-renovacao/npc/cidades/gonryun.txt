alberta,245,69,3	script	Apreciador de Kunlun#gon	776,{
	mes "[Wa Bai Hu]";
	mes "Bom dia~";
	mes "Deixem-me convidá-los para";
	mes "visitar a minha terra natal, Kunlun.";
	mes "É uma honra guiar convidados";
	mes "tão distintos, naturais do";
	mes "reino de Rune-Midgard";
	next;
	switch(select("Sobre Kunlun:Visitar Kunlun:Cancelar")) {
	case 1:
		mes "[Wa Bai Hu]";
		mes "Kunlun é um lugar maravilhoso,";
		mes "com uma história rica e tradições";
		mes "próprias. Acredito também que";
		mes "Kunlun é o melhor lugar para se";
		mes "apreciar uma bela vista.";
		next;
		mes "[Wa Bai Hu]";
		mes "Quando você chegar no porto";
		mes "de Kunlun, poderá ver";
		mes "réplicas em miniatura de";
		mes "prédios de Alberta";
		mes "e Prontera.";
		next;
		mes "[Wa Bai Hu]";
		mes "Depois de aproveitar um bom";
		mes "passeio, passe pela bonita";
		mes "coluna de luz que te levará";
		mes "acima das nuvens, até a";
		mes "vila de Kunlun.";
		next;
		mes "[Wa Bai Hu]";
		mes "Já ouvi dizer que no";
		mes "continente de Rune-Midgarts";
		mes "existe outra cidade que é";
		mes "mantida suspensa no ar por";
		mes "um poder antigo e misterioso...";
		next;
		mes "[Wa Bai Hu]";
		mes "Bem, Kunlun também flutua";
		mes "no ar, mas sem nenhuma";
		mes "tecnologia ou fonte de energia.";
		mes "Consideramos que nossa terra";
		mes "é especialmente abençoada...";
		next;
		mes "[Wa Bai Hu]";
		mes "Quando estiver em Kunlun,";
		mes "não deixe de experimentar nossas";
		mes "especialidades como o pêssego do céu.";
		next;
		mes "[Wa Bai Hu]";
		mes "E é melhor você se preparar";
		mes "se estiver planejando uma";
		mes "visita à dungeon de Kunlun.";
		mes "Não é um lugar seguro!";
		next;
		mes "[Wa Bai Hu]";
		mes "Se você está interessado";
		mes "em visitar Kunlun, venha";
		mes "logo me dizer. É um prazer";
		mes "serví-lo, honroso convidado.";
		close;
	case 2:
		mes "[Wa Bai Hu]";
		mes "Excelente escolha! Estou feliz";
		mes "por tê-lo como convidado~";
		mes "Entretanto, uma pequena taxa";
		mes "é requerida para embarcar.";
		next;
		mes "[Wa Bai Hu]";
		mes "Pedimos que pague 10.000 zenys";
		mes "antes de viajar. Essa taxa já";
		mes "inclui seu retorno a Alberta.";
		mes "Estou pronto para levá-lo a";
		mes "Kunlun quando quiser.";
		next;
		mes "[Wa Bai Hu]";
		mes "Você gostaria de embarcar?";
		next;
		if (select("Para Kunlun~!:Não.") == 1) {
			if (Zeny > 9999) {
				mes "[Wa Bai Hu]";
				mes "Obrigado, levarei você imediatamente.";
				close2;
				set zeny,zeny-10000;
				warp "gon_fild01",258,82;
				end;
			}
			mes "[Wa Bai Hu]";
			mes "Sinto muito, você precisa de";
			mes "10 000 zenys para viajar à Kunlun.";
			mes "Certifique-se de ter dinheiro";
			mes "suficiente em suas mãos. Obrigado";
			mes "e retorne quando quiser!";
			close;
		}
		mes "[Wa Bai Hu]";
		mes "Entendo. Mas, se mudar de";
		mes "ideia, por favor me avise.";
		mes "Seria um enorme prazer";
		mes "servi-lo, honroso";
		mes "viajante.";
		close;
	case 3:
		mes "[Wa Bai Hu]";
		mes "Entendo. Entretanto, se mudar";
		mes "de ideia, me avise.";
		mes "Será um prazer serví-lo.";
		close;
	}
}
